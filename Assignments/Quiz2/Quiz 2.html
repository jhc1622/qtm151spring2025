def fn_stats(TotalStats):
    if TotalStats > 85:
        classification = "Top Performer"
    else:
        classification = "Non Top-Performer"
    return classification

fifa["player_classification"] = fifa["TotalStats"].apply(fn_stats)
display(fifa)
